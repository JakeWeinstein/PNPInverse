**Re your R4 point 2** (k0 ladder overshoots target):
**Accept fully.** Programmatic ladder:
```python
def _adaptive_ladder(start: float, target: float, *,
                     factor_init: float = 32.0,
                     factor_min: float = 4.0):
    """Yield k0 values from start to target (inclusive), with
    geometric step factor_init, never overshooting target."""
    cur = start
    while cur < target:
        next_val = min(cur * factor_init, target)
        yield next_val
        cur = next_val
```
  Same logic for dt ramp (programmatic, capped at 0.25).

```python
def assert_picard_residual_log_consistent(
    R_picard: float, R_residual_per_area: float, *,
    log_tol: float = 0.05,
    abs_floor: float = 1e-10,
    tag: str = "",
):
    if abs(R_picard) < abs_floor or abs(R_residual_per_area) < abs_floor:
        # Tiny rates: use absolute tolerance; sign-sensitive.
        assert abs(R_picard - R_residual_per_area) < abs_floor, ...
        return
    log_ratio = math.log10(abs(R_residual_per_area) / abs(R_picard))
    assert abs(log_ratio) < log_tol, (
        f"Picard ({R_picard:.4e}) and residual ({R_residual_per_area:.4e}) "
        f"disagree by {log_ratio:+.3f} log-decades; > tol {log_tol}; "
        f"tag={tag}"
    )
```
